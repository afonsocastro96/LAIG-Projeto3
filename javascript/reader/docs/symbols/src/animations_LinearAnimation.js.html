1 /**
  2  * LinearAnimation constructor.
  3  * @constructor
  4  * @param id {String} The animation's id.
  5  * @param span {Float} The animation's span.
  6  * @param controlPoints {Array} The animation's control points.
  7  */
  8 function LinearAnimation(id, span, controlPoints) {
  9     Animation.call(this, id, span, "linear");
 10 
 11     this.controlPoints = controlPoints;
 12 
 13     this.init();
 14 }
 15 
 16 LinearAnimation.prototype = Object.create(Animation.prototype);
 17 LinearAnimation.prototype.constructor = LinearAnimation;
 18 
 19 /**
 20  * Initiates the animation.
 21  */
 22 LinearAnimation.prototype.init = function() {
 23     if (this.controlPoints.length == 1)
 24         return;
 25     var distance = 0;
 26     this.translations = new Array(this.controlPoints.length - 1);
 27     this.rotations = new Array(this.controlPoints.length - 1);
 28 
 29     for (var i = 0; i < this.controlPoints.length - 1; ++i) {
 30         var vector = vec3.create();
 31         vec3.sub(vector, this.controlPoints[i + 1], this.controlPoints[i]);
 32         this.translations[i] = vector;
 33 
 34         var projectionXZ = vec3.fromValues(vector[0], 0, vector[2]);
 35 
 36         if (vec3.length(projectionXZ) > 0) {
 37             var sign = projectionXZ[2] < 0 ? 1 : -1;
 38             this.rotations[i] = sign * Math.acos(vec3.dot(projectionXZ, vec3.fromValues(1, 0, 0))/ vec3.length(projectionXZ));
 39         } else {
 40             this.rotations[i] = (i == 0 ? 0 : this.rotations[i - 1]);  
 41         }
 42     
 43         distance += vec3.length(vector);
 44     }
 45 
 46     var velocity = distance / this.span;
 47 
 48     this.controlPointsTime = new Array(this.controlPoints.length);
 49     this.controlPointsTime[0] = 0;
 50 
 51     this.controlPointsSpan = new Array(this.controlPoints.length - 1);
 52 
 53     for (var i = 1; i < this.controlPoints.length; ++i) {    
 54         this.controlPointsTime[i] = this.controlPointsTime[i - 1] +
 55                                vec3.length(this.translations[i-1]) / velocity;
 56         this.controlPointsSpan[i-1] = this.controlPointsTime[i] - this.controlPointsTime[i-1]; 
 57     }
 58 }
 59 
 60 /** 
 61  * Calculates the matrix while the animation is running.
 62  * @param t {Float} The current time.
 63  * @return {mat4} Animation matrix.
 64  */
 65 LinearAnimation.prototype.calculateMatrix = function(t) {
 66     var matrix = mat4.create();
 67     mat4.identity(matrix);
 68     
 69     if (t < 0)
 70         return matrix;
 71 
 72     t = Math.min(t, this.span);
 73 
 74     if (this.controlPoints.length == 1) {
 75         mat4.translate(matrix, matrix, this.controlPoints[0]);
 76         return matrix;
 77     }
 78    
 79     var index;
 80     for (index = this.controlPointsTime.length - 1; index > 0; --index)
 81         if (this.controlPointsTime[index] < t)
 82             break;
 83 
 84     var tScale = (t - this.controlPointsTime[index]) / this.controlPointsSpan[index];
 85     var position = vec3.clone(this.controlPoints[index]);
 86     var translation_amount = vec3.create();
 87     vec3.scale(translation_amount, this.translations[index], tScale);
 88     vec3.add(position, position, translation_amount); 
 89 
 90     mat4.translate(matrix, matrix, position);
 91     mat4.rotateY(matrix,matrix,this.rotations[index]);
 92     
 93     return matrix;
 94 }
