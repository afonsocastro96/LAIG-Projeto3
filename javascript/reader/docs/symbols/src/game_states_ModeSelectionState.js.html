1 /**
  2  * ModeSelectionState constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which the ModeSelectionState belongs.
  5  */
  6 function ModeSelectionState() {
  7 	GameState.call(this);
  8 }
  9 
 10 ModeSelectionState.prototype = Object.create(GameState.prototype);
 11 ModeSelectionState.prototype.constructor = BoardSelectionState;
 12 
 13 /**
 14 * ModeSelectionState initializer.
 15 */
 16 ModeSelectionState.prototype.init = function(gameSet) {
 17 	var gameState = this;
 18 	
 19 	this.displayHUD = this.displayGameModesHUD;
 20 	
 21 	this.humanVsHumanButton = new Marker(gameSet.scene);
 22 	this.humanVsHumanButton.setText("Human VS Human");
 23 	this.humanVsHumanPick = {
 24 		onPick : function() {
 25 			gameState.onPickMode(gameSet, Connection.humanVsHumanMode);
 26 		}
 27 	};
 28 	
 29 	this.humanVsMachineButton = new Marker(gameSet.scene);
 30 	this.humanVsMachineButton.setText("Human VS Machine");
 31 	this.humanVsMachinePick = {
 32 		onPick : function() {
 33 			gameState.onPickMode(gameSet, Connection.humanVsMachineMode);
 34 		}
 35 	};
 36 	
 37 	this.machineVsMachineButton = new Marker(gameSet.scene);
 38 	this.machineVsMachineButton.setText("Machine VS Machine");
 39 	this.machineVsMachinePick = {
 40 		onPick : function() {
 41 			gameState.onPickMode(gameSet, Connection.machineVsMachineMode);
 42 		}
 43 	};
 44 	
 45 	this.easyDifficultyButton = new Marker(gameSet.scene);
 46 	this.easyDifficultyButton.setText("Easy");
 47 	this.easyDifficultyPick = {
 48 		onPick : function() {
 49 			gameState.onPickDifficulty(gameSet, Connection.easyDifficulty);
 50 		}
 51 	}
 52 	
 53 	this.hardDifficultyButton = new Marker(gameSet.scene);
 54 	this.hardDifficultyButton.setText("Hard");
 55 	this.hardDifficultyPick = {
 56 		onPick : function() {
 57 			gameState.onPickDifficulty(gameSet, Connection.hardDifficulty);
 58 		}
 59 	}
 60 }
 61 
 62 /**
 63 * Display function used to render this object.
 64 */
 65 ModeSelectionState.prototype.display = function(gameSet) {
 66 	gameSet.displayStatic();
 67 }
 68 
 69 /**
 70 * Displays the state's HUD.
 71 */
 72 ModeSelectionState.prototype.displayGameModesHUD = function(gameSet) {
 73 	gameSet.scene.pushMatrix();
 74 		gameSet.scene.translate(0,3,-20);
 75 		gameSet.scene.scale(0.5, 0.5, 0.5);
 76 
 77 		gameSet.scene.registerNextPick(this.humanVsHumanPick);
 78 		this.humanVsHumanButton.display();
 79 	gameSet.scene.popMatrix();
 80 	
 81 	gameSet.scene.pushMatrix();
 82 		gameSet.scene.translate(0, 0, -20);
 83 		gameSet.scene.scale(0.5, 0.5, 0.5);
 84 		
 85 		gameSet.scene.registerNextPick(this.humanVsMachinePick);
 86 		this.humanVsMachineButton.display();
 87 	gameSet.scene.popMatrix();
 88 	
 89 	gameSet.scene.pushMatrix();
 90 		gameSet.scene.translate(0,-3,-20);
 91 		gameSet.scene.scale(0.5, 0.5, 0.5);
 92 
 93 		gameSet.scene.registerNextPick(this.machineVsMachinePick);
 94 		this.machineVsMachineButton.display();
 95 	gameSet.scene.popMatrix();
 96 	
 97 	gameSet.scene.clearPickRegistration();
 98 }
 99 
100 /**
101 * Displays the difficulties HUD.
102 */
103 ModeSelectionState.prototype.displayDifficultiesHUD = function(gameSet) {
104 	gameSet.scene.pushMatrix();
105 		gameSet.scene.translate(0,1,-20);
106 		gameSet.scene.scale(0.75, 0.75, 0.75);
107 
108 		gameSet.scene.registerNextPick(this.easyDifficultyPick);
109 		this.easyDifficultyButton.display();
110 	gameSet.scene.popMatrix();
111 	
112 	gameSet.scene.pushMatrix();
113 		gameSet.scene.translate(0,-1,-20);
114 		gameSet.scene.scale(0.75, 0.75, 0.75);
115 		
116 		gameSet.scene.registerNextPick(this.hardDifficultyPick);
117 		this.hardDifficultyButton.display();
118 	gameSet.scene.popMatrix();
119 	
120 	gameSet.scene.clearPickRegistration();
121 }
122 
123 /**
124 * Function called when a game mode is picked.
125 */
126 ModeSelectionState.prototype.onPickMode = function(gameSet, gameMode) {
127 	var gameState = this;
128 	
129 	Connection.gamemode(gameSet,
130 	function (gameSet, request) {
131 		gameState.modeSelected(gameSet, request, gameMode);
132 	},
133 	gameMode);
134 }
135 
136 /**
137 * Function called when a difficulty is picked.
138 */
139 ModeSelectionState.prototype.onPickDifficulty = function(gameSet, difficulty) {
140 	Connection.setDifficulty(gameSet, this.difficultySelected, difficulty);
141 }
142 
143 /**
144 * Function called to change the state when a mode is selected.
145 */
146 ModeSelectionState.prototype.modeSelected = function(gameSet, request, gameMode) {	
147 	if (gameMode == Connection.humanVsHumanMode) {
148 		gameSet.setState(new TowerSelectionState());
149 		return;
150 	}
151 	
152 	this.displayHUD = this.displayDifficultiesHUD;
153 }
154 
155 /**
156 * Function called to change the state when a difficulty is selected.
157 */
158 ModeSelectionState.prototype.difficultySelected = function(gameSet, request) {
159 	gameSet.setState(new TowerSelectionState());
160 }
