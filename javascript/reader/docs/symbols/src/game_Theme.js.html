1 /**
  2  * Theme constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which the Theme belongs.
  5  * @param path {string} The path to the LSX file from this theme.
  6  * @param id {string} Identification of the theme
  7  */
  8 function Theme(scene, path, id) {
  9 	this.scene = scene;
 10 	this.path = path;
 11 	this.loaded = false;
 12 	var theme = this;
 13 	this.graph = new LSXSceneGraph(path, scene, function() {theme.onGraphLoaded();});
 14 	this.primitives = [];
 15 	this.id = id;
 16 }
 17 
 18 Theme.prototype = Object.create(Object.prototype);
 19 Theme.prototype.constructor = Theme;
 20 
 21 /**
 22  * Function to be called when the theme's SceneGraph is done loading
 23  */
 24 Theme.prototype.onGraphLoaded = function() 
 25 {
 26     for (key in this.graph.leaves) {
 27     	var leaf = this.graph.leaves[key];
 28     	switch (leaf.type) {
 29     		case "rectangle":
 30     			this.primitives[key] = new MyRectangle(this.scene, leaf.v1x, leaf.v1y, leaf.v2x, leaf.v2y);
 31     			break;
 32     		case "triangle":
 33     			this.primitives[key] = new MyTriangle(this.scene, leaf.v1[0], leaf.v1[1], leaf.v1[2], leaf.v2[0], leaf.v2[1], leaf.v2[2], leaf.v3[0], leaf.v3[1], leaf.v3[2]);
 34     			break;
 35     		case "cylinder":
 36 				this.primitives[key] = new MyCylinder(this.scene, leaf.height, leaf.bottomRadius, leaf.topRadius, leaf.slices, leaf.stacks);
 37 				break;
 38 			case "sphere":
 39 				this.primitives[key] = new MySphere(this.scene, leaf.radius, leaf.slices, leaf.stacks);
 40 				break;
 41 			case "plane":
 42 				this.primitives[key] = new MyPlane(this.scene, leaf.parts);
 43 				break;
 44 			case "patch":
 45 				this.primitives[key] = new MyPatch(this.scene, leaf.order, leaf.partsU, leaf.partsV, leaf.controlPoints);
 46 				break;
 47 			case "vehicle":
 48 				this.primitives[key] = new MyVehicle(this.scene);
 49 				break;
 50 			case "terrain":
 51 				this.primitives[key] = new MyTerrain(this.scene, leaf.textureUrl, leaf.heightmapUrl);
 52 				break;
 53 			case "gameset":
 54 				this.primitives[key] = this.scene.gameSet;
 55 				break;
 56 			default:
 57 				console.warn("Unknown primitive: " + leaf.type);
 58 				break;
 59     	}
 60     }
 61 
 62 	this.loaded = true;
 63 };
