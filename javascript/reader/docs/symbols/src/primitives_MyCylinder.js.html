1 /**
  2  * My Cylinder constructor.
  3  * @constructor
  4  * @param scene {CGFscene} to which this cylinder belongs.
  5  * @param height {Float} Cylinder height
  6  * @param bRadius {Float} Cylinder bottom radius
  7  * @param tRadius {Float} Cylinder top radius
  8  * @param slices {Integer} Number of divisions of each circle
  9  * @param stacks {Integer} Number of divisions of the cylinder
 10  */
 11 function MyCylinder(scene, height, bRadius, tRadius, stacks, slices) {
 12  	CGFobject.call(this,scene);
 13 	
 14 	this.height = height;
 15 
 16 	this.slices=slices;
 17 	this.stacks=stacks;
 18 
 19 	this.bRadius = bRadius;
 20 	this.tRadius = tRadius;
 21 
 22  	this.initBuffers();
 23 }
 24 
 25 MyCylinder.prototype = Object.create(CGFobject.prototype);
 26 MyCylinder.prototype.constructor = MyCylinder;
 27 
 28 /**
 29  * Method in which the geometry of the cylinder is defined.
 30  */
 31 MyCylinder.prototype.initBuffers = function() {
 32     var dTheta = 2*Math.PI/this.slices;
 33 	var dRadius = (this.tRadius - this.bRadius) / this.stacks;
 34 	var dHeight = this.height / this.stacks;
 35 
 36 	this.vertices=[];
 37  	this.normals=[];
 38 	this.texCoords=[];
 39 
 40 	var vecNormal = vec3.create();
 41 
 42  	for (var stack = 0; stack <= this.stacks; ++stack) {
 43  		var radius = this.bRadius + dRadius * stack;
 44  		var height = dHeight * stack;
 45  		for (var slice = 0; slice <= this.slices; ++slice) {
 46  			var theta = dTheta * slice;
 47  			this.vertices.push(radius * Math.cos(theta),radius*Math.sin(theta),height);
 48 
 49 			var vecDiffSlice = vec3.fromValues(radius*(-Math.sin(theta))*dTheta, radius*Math.cos(theta)*dTheta, 0);
 50 			var vecDiffStack = vec3.fromValues(Math.cos(theta)*dRadius, Math.sin(theta)*dRadius, dHeight);
 51 			vec3.cross(vecNormal, vecDiffSlice, vecDiffStack);
 52 			vec3.normalize(vecNormal, vecNormal);
 53 
 54  			this.normals.push(vecNormal[0],vecNormal[1],vecNormal[2]);
 55  			this.texCoords.push(slice/this.slices, 1-stack/this.stacks);
 56  		}
 57  	}
 58 
 59  	this.indices=[];
 60  	for (var stack = 0; stack < this.stacks; ++stack) {
 61  		for (var slice = 0; slice < this.slices; ++slice) {
 62  			this.indices.push(stack * (this.slices+1) + slice, stack * (this.slices+1) + slice + 1, (stack+1) * (this.slices+1) + slice + 1);
 63  			this.indices.push(stack * (this.slices+1) + slice, (stack+1) * (this.slices+1) + slice + 1, (stack+1) * (this.slices+1) + slice);
 64  		}
 65  	}
 66 	
 67     this.primitiveType=this.scene.gl.TRIANGLES;
 68 	this.initGLBuffers();
 69  };
