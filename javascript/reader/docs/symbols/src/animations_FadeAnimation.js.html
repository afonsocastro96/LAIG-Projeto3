1 /**
  2  * FadeAnimation constructor.
  3  * @constructor
  4  * @param span {Integer} Animation duration in miliseconds.
  5  * @param startScale {Float} Initial value of alpha scaling.
  6  * @param endScale {Float} Final value of alpha scaling.
  7  */
  8 function FadeAnimation(span, startScale, endScale) {
  9 	this.span = span;
 10 	this.startScale = startScale;
 11 	this.endScale = endScale;
 12 	this.scaleInc = (this.endScale - this.startScale) / this.span;
 13 }
 14 
 15 FadeAnimation.prototype = Object.create(Object.prototype);
 16 FadeAnimation.prototype.constructor = FadeAnimation;
 17 
 18 
 19 /**
 20  * Calculate the alpha scaling.
 21  * @param t {Integer} Time (in miliseconds) occurred since the beginning of the animation.
 22  */
 23 FadeAnimation.prototype.getScaling = function(t) {
 24 	if (t < 0) {
 25 		return this.startScale;
 26 	}
 27 	if (t > this.span) {
 28 		return this.endScale;
 29 	}
 30 	
 31 	return this.startScale + this.scaleInc * t;
 32 }
