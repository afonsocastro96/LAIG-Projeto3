1 /**
  2  * CircularAnimation constructor.
  3  * @constructor
  4  * @param id {String} The animation's id.
  5  * @param span {Float} The animation's span.
  6  * @param center {vec3} The animation's center of rotation.
  7  * @param startAngle {Float} The animation's start angle.
  8  * @param rotAngle {Float} The animation's rotation.
  9  * @param radius {Float} The animation's radius.
 10  */
 11 function CircularAnimation(id, span, center, startAngle, rotAngle, radius) {
 12     Animation.call(this, id, span, "circular");
 13     this.id = id;
 14     this.span = span;
 15     this.center = center;
 16     this.startAngle = startAngle;
 17     this.rotAngle = rotAngle;
 18     this.radius = radius;
 19     this.init();
 20 }
 21 
 22 CircularAnimation.prototype = Object.create(Animation.prototype);
 23 CircularAnimation.prototype.constructor = CircularAnimation;
 24 
 25 /**
 26  * Initiates the animation.
 27  */
 28 CircularAnimation.prototype.init = function() {
 29 	this.initialTransformation = mat4.create();
 30 	mat4.identity(this.initialTransformation);
 31 	mat4.rotateY(this.initialTransformation, this.initialTransformation, this.startAngle);
 32 	mat4.translate(this.initialTransformation, this.initialTransformation,
 33 				   vec3.fromValues(this.radius, 0, 0));
 34 	
 35 	mat4.rotateY(this.initialTransformation, this.initialTransformation,
 36 				this.rotAngle > 0 ? Math.PI / 2 : - Math.PI / 2);
 37 }
 38 
 39 /** 
 40  * Calculates the matrix while the animation is running.
 41  * @param t {Float} The current time.
 42  * @return {mat4} Animation matrix.
 43  */
 44 CircularAnimation.prototype.calculateMatrix = function(t) {
 45 
 46 	t = Math.min(t, this.span);
 47 
 48 	var matrix = mat4.create();
 49 	mat4.identity(matrix);
 50 	
 51 	if (t < 0)
 52 		return matrix;
 53 
 54 	mat4.translate(matrix, matrix, this.center);
 55 
 56 	var rot = this.rotAngle * (t / this.span);
 57 	mat4.rotateY(matrix, matrix, rot);
 58 	mat4.multiply(matrix, matrix, this.initialTransformation);
 59 
 60 	return matrix;
 61 }
