1 /**
  2  * TileStack constructor.
  3  * @constructor
  4  * @param scene {CGFscene} scene this tile stack belongs to.
  5  */
  6 function TileStack(scene) {
  7     this.scene = scene;
  8 
  9     this.blackCirclesStack = [];
 10     this.blackSquaresStack = [];
 11     this.whiteCirclesStack = [];
 12     this.whiteSquaresStack = [];
 13 }
 14 
 15 TileStack.prototype = Object.create(Object.prototype);
 16 TileStack.prototype.constructor = TileStack;
 17 
 18 /**
 19  * Add tile to respective stack.
 20  * @param tile {Tile} tile to add to stack.
 21  */
 22 TileStack.prototype.addTile = function(tile) {
 23 	if (tile instanceof BlackCircleTile)
 24 		this.blackCirclesStack.push(tile);
 25 	else if (tile instanceof BlackSquareTile)
 26 		this.blackSquaresStack.push(tile);
 27 	else if (tile instanceof WhiteCircleTile)
 28 		this.whiteCirclesStack.push(tile);
 29 	else if (tile instanceof WhiteSquareTile)
 30 		this.whiteSquaresStack.push(tile);
 31 }
 32 
 33 /**
 34  * Remove tile from respective stack.
 35  * @param tile {Tile} tile to remove.
 36  */
 37 TileStack.prototype.removeTile = function(tile) {
 38 	if (tile instanceof BlackCircleTile)
 39 		this.blackCirclesStack.pop();
 40 	else if (tile instanceof BlackSquareTile)
 41 		this.blackSquaresStack.pop();
 42 	else if (tile instanceof WhiteCircleTile)
 43 		this.whiteCirclesStack.pop();
 44 	else if (tile instanceof WhiteSquareTile)
 45 		this.whiteSquaresStack.pop();
 46 }
 47 
 48 /**
 49  * Display tile stacks.
 50  */
 51 TileStack.prototype.display = function() {
 52     this.scene.pushMatrix();
 53 		this.scene.translate(-1.5,0.1,0);
 54 
 55 		this.scene.pushMatrix();
 56 			for (var i = 0; i < this.blackCirclesStack.length; ++i) {
 57 				this.blackCirclesStack[i].display();
 58 				this.scene.translate(0,0.2,0);
 59 			}
 60 		this.scene.popMatrix();
 61 		
 62 		this.scene.pushMatrix();
 63 			this.scene.translate(1,0,0);
 64 			for (var i = 0; i < this.blackSquaresStack.length; ++i) {
 65 				this.blackSquaresStack[i].display();
 66 				this.scene.translate(0,0.2,0);
 67 			}
 68 		this.scene.popMatrix();
 69 		
 70 		this.scene.pushMatrix();
 71 			this.scene.translate(2,0,0);
 72 			for (var i = 0; i < this.whiteCirclesStack.length; ++i) {
 73 				this.whiteCirclesStack[i].display();
 74 				this.scene.translate(0,0.2,0);
 75 			}
 76 		this.scene.popMatrix();
 77 	
 78 		this.scene.pushMatrix();
 79 			this.scene.translate(3,0,0);
 80 			for (var i = 0; i < this.whiteSquaresStack.length; ++i) {
 81 				this.whiteSquaresStack[i].display();
 82 				this.scene.translate(0,0.2,0);
 83 			}
 84 		this.scene.popMatrix();
 85     
 86 	this.scene.popMatrix();
 87 }
