1 /**
  2  * MyTriangle constructor.
  3  * @constructor
  4  * @param scene {CGFscene} The scene to which this triangle belongs.
  5  * @param x1 {Float} First point x coordinate
  6  * @param y1 {Float} First point y coordinate
  7  * @param z1 {Float} First point z coordinate
  8  * @param x2 {Float} Second point x coordinate
  9  * @param y2 {Float} Second point y coordinate
 10  * @param z2 {Float} Second point z coordinate
 11  * @param x3 {Float} Third point x coordinate
 12  * @param y3 {Float} Third point y coordinate
 13  * @param z3 {Float} Third point z coordinate
 14  */
 15 function MyTriangle(scene, x1,y1,z1,x2,y2,z2,x3,y3,z3){
 16     CGFobject.call(this,scene);
 17 
 18 	this.v1 = vec3.fromValues(x1, y1, z1);
 19 	this.v2 = vec3.fromValues(x2, y2, z2);
 20 	this.v3 = vec3.fromValues(x3, y3, z3);
 21   
 22     this.initBuffers();
 23 }
 24 
 25 MyTriangle.prototype = Object.create(CGFobject.prototype);
 26 MyTriangle.prototype.constructor = MyTriangle;
 27 
 28 /**
 29  * Method in which the geometry of the cylinder is defined.
 30  */
 31 MyTriangle.prototype.initBuffers = function() {
 32 
 33     this.vertices = [
 34     	this.v1[0], this.v1[1], this.v1[2],
 35     	this.v2[0], this.v2[1], this.v2[2],
 36     	this.v3[0], this.v3[1], this.v3[2]
 37     ];
 38 
 39     this.indices = [0,1,2];
 40 
 41 	var AB = vec3.create();
 42 	vec3.sub(AB, this.v2, this.v1);
 43 	var AC = vec3.create();
 44 	vec3.sub(AC, this.v3, this.v1);
 45 	var BC = vec3.create();
 46 	vec3.sub(BC, this.v3, this.v2);
 47 
 48 	var N = vec3.create();
 49 	vec3.cross(N, AB, BC);
 50 	vec3.normalize(N, N);
 51 
 52 	this.normals = [
 53 		N[0], N[1], N[2],
 54 		N[0], N[1], N[2],
 55 		N[0], N[1], N[2],
 56     ];
 57 
 58     var tC = (vec3.sqrLen(AB) + vec3.sqrLen(AC) - vec3.sqrLen(BC))/ (2 * vec3.length(AB));
 59 	var sC = Math.sqrt(vec3.sqrLen(AC) - tC * tC);
 60 	this.nonScaledTexCoords = [
 61 		0,0,
 62 		vec3.length(AB),0,
 63 		sC, tC
 64 	];
 65 
 66 	this.texCoords = this.nonScaledTexCoords.slice(0);
 67 
 68     this.primitiveType=this.scene.gl.TRIANGLES;
 69 
 70 	this.initGLBuffers();
 71 }
 72 
 73 /**
 74  * texCoords scaling.
 75  * @param ampS {Float} Scaling along the S axis.
 76  * @param ampT {Float} Scaling along the T axis.
 77  */
 78 MyTriangle.prototype.scaleTexCoords = function(ampS, ampT) {
 79 	for (var i = 0; i < this.texCoords.length; i += 2) {
 80 		this.texCoords[i] = this.nonScaledTexCoords[i] / ampS;
 81 		this.texCoords[i + 1] = this.nonScaledTexCoords[i+1] / ampT;
 82 	}
 83 	this.updateTexCoordsGLBuffers();
 84 }
