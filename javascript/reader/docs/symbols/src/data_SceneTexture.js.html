1 /**
  2  * SceneGraph texture constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which this texture belongs.
  5  * @param url {string} Path to the texture image.
  6  * @param id {string} Texture identification.
  7  */
  8 function SceneTexture(scene, url, id) {
  9     CGFtexture.call(this, scene, url);
 10     this.id = id;
 11     this.amplifyFactor = {s: 1,
 12                           t: 1};
 13 };
 14 
 15 SceneTexture.prototype = Object.create(CGFtexture.prototype);
 16 SceneTexture.prototype.constructor = SceneTexture;
 17 
 18 /**
 19  * AmplifyFactor setter.
 20  * @param s {number} S value to set.
 21  * @param t {number} T value to set.
 22  */
 23 SceneTexture.prototype.setAmplifyFactor = function(s, t) {
 24     this.amplifyFactor.s = s;
 25     this.amplifyFactor.t = t;
 26 }
