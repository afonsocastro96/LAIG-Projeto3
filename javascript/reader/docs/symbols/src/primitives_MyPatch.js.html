1 /**
  2  * MyPatch constructor.
  3  * @constructor
  4  * @param scene {CGFscene} The scene to which this cylinder belongs.
  5  * @param order {Integer} The order of the patch.
  6  * @param partsU {Integer} Number of parts along U.
  7  * @param partsV {Integer} Number of parts along V.
  8  * @param controlPoints {Array} The patch's control points.
  9  */
 10 function MyPatch(scene, order, partsU, partsV, controlPoints){
 11 
 12 	var controlPointsSurface = [];
 13 	var nArrays = Math.sqrt(controlPoints.length);
 14 
 15 	for(var i = 0; i < nArrays; ++i){
 16 		var controlPointsTemp = [];
 17 		for(var j = 0; j < nArrays; ++j){
 18 			controlPointsTemp.push(controlPoints[i*nArrays+j]);
 19 		}
 20 		controlPointsSurface.push(controlPointsTemp);
 21 	}
 22 
 23 
 24 	var knots = [];
 25 
 26 	for(var i = 0; i < (order+1); ++i){
 27 		knots.push(0);
 28 	}
 29 
 30 	for(var i = 0; i < (order+1); ++i){
 31 		knots.push(1);
 32 	}
 33 
 34 	var nurbsSurface = new CGFnurbsSurface(order, order, knots, knots, controlPointsSurface);
 35 	getSurfacePoint = function(u, v) {
 36 		return nurbsSurface.getPoint(u, v);
 37 	};
 38 
 39     CGFnurbsObject.call(this,scene, getSurfacePoint, partsU, partsV);
 40 
 41     this.fixTexCoords();
 42 }
 43 
 44 MyPatch.prototype = Object.create(CGFnurbsObject.prototype);
 45 MyPatch.prototype.constructor = MyPatch;
 46 
 47 /**
 48  * fixes the texCoords along the s axis.
 49  */
 50 MyPatch.prototype.fixTexCoords = function() {
 51 	for (var i = 0; i < this.texCoords.length; i += 2)
 52 		this.texCoords[i] = 1 - this.texCoords[i];
 53 
 54 	this.initGLBuffers();
 55 }
 56

 58  * texCoords scaling (no effect).
 59  */
 60 MyPatch.prototype.scaleTexCoords = function() {}
