1 /**
  2  * GameBoard constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which the GameBoard belongs.
  5  */
  6 function GameBoard(scene) {
  7     this.scene = scene;
  8     this.emptyTile = new EmptyTile();
  9 }
 10 
 11 GameBoard.prototype = Object.create(Object.prototype);
 12 GameBoard.prototype.constructor = GameBoard;
 13 
 14 /**
 15 * GameBoard initializer.
 16 */
 17 GameBoard.prototype.init = function(boardSchema) {
 18 	this.board = [];
 19 	for(var row = 0; row < boardSchema.length; ++row){
 20 		var currentRow = [];
 21 		for(var col = 0; col < boardSchema[row].length; ++col){
 22 			switch(boardSchema[row][col][0]){
 23 				case 0:
 24 					currentRow.push(new EmptyTile());
 25 					break;
 26 				case 1:
 27 					if(boardSchema[row][col][1] == 3)
 28 						currentRow.push(new WhiteCircleTile(this.scene));
 29 					else
 30 						currentRow.push(new WhiteSquareTile(this.scene));
 31 					break;
 32 				case 2:
 33 					if(boardSchema[row][col][1] == 3)
 34 						currentRow.push(new BlackCircleTile(this.scene));
 35 					else
 36 						currentRow.push(new BlackSquareTile(this.scene));
 37 					break;
 38 				default:
 39 					break;
 40 			}
 41 		}
 42 		this.board.push(currentRow);
 43 	}
 44 	
 45     this.numRows = this.board.length;
 46     this.numCols = this.board[0].length;
 47 }
 48 
 49 /**
 50 * Display function used to render this object.
 51 */
 52 GameBoard.prototype.display = function() {
 53     this.scene.pushMatrix();
 54     this.scene.translate(-this.numCols/2 + 0.5, 0.1, -this.numRows/2 + 0.5);
 55 
 56     for (var row = 0; row < this.numRows; ++row) {
 57         for (var col = 0; col < this.numCols; ++col) {
 58             this.board[row][col].display();
 59             this.scene.translate(1,0,0);
 60         }
 61         this.scene.translate(-this.numCols, 0, 1);
 62     }
 63 
 64     this.scene.popMatrix();
 65 }
 66 /**
 67 * Moves a tile.
 68 */
 69 GameBoard.prototype.moveTile = function(startRow, startCol, endRow, endCol) {
 70     this.board[endRow][endCol] = this.board[startRow][startCol];
 71     this.board[startRow][startCol] = this.emptyTile;
 72 }
 73 
 74 /**
 75 * Removes a tile.
 76 */
 77 GameBoard.prototype.removeTile = function(row, col) {
 78 	this.board[row][col] = this.emptyTile;
 79 }
 80 
 81 /**
 82 * Returns a tile.
 83 * @returns {BoardTile} The tile.
 84 */
 85 GameBoard.prototype.getTile = function(row, col) {
 86 	return this.board[row][col];
 87 }
 88 
 89 /**
 90 * Adds a tile to the board.
 91 */
 92 GameBoard.prototype.addTile = function(row, col, tile) {
 93     this.board[row][col] = tile;
 94 }
 95 
 96 /**
 97 * Function used for coordinates to array position conversion.
 98 * @returns {int} The array position.
 99 */
100 GameBoard.prototype.getBoardCoordinates = function(row, col) {
101 	return [-(this.numCols-1)/2 + col, 0.2, -(this.numRows-1)/2 + row];
102 }
