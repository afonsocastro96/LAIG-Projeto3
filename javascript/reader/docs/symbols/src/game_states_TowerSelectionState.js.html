1 /**
  2  * TowerSelectionState constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which the TowerSelectionState belongs.
  5  */
  6 function TowerSelectionState() {
  7 	GameState.call(this);
  8 }
  9 
 10 TowerSelectionState.prototype = Object.create(GameState.prototype);
 11 TowerSelectionState.prototype.constructor = BoardSelectionState;
 12 
 13 /**
 14 * TowerSelectionState initializer.
 15 */
 16 TowerSelectionState.prototype.init = function(gameSet) {
 17 	gameSet.towers = [];
 18 	var gameState = this;
 19 	Connection.setupTowers(gameSet, function (target, request) { gameState.towerSet(target, request); });
 20 	
 21 	this.selectionPanel = new MyPlane(gameSet.scene, 100);
 22 	this.selectionPanelAppearance = new CGFappearance(gameSet.scene);
 23 	this.selectionPanelAppearance.setAmbient(0.0, 0.0, 0.3, 0.1);
 24 	this.selectionPanelAppearance.setDiffuse(0.0, 0.0, 0.8, 0.1);
 25 	this.selectionPanelAppearance.setSpecular(0.0, 0.0, 0.8, 0.1);
 26 	
 27 	this.marker = new Marker(gameSet.scene);
 28 	this.positions = [];
 29 }
 30 
 31 /**
 32 * Displays the state's HUD.
 33 */
 34 TowerSelectionState.prototype.displayHUD = function(gameSet) {
 35 	gameSet.scene.pushMatrix();
 36 		gameSet.scene.translate(0, -3.5, -20);
 37 		gameSet.scene.scale(0.5, 0.5, 0.5);
 38 		
 39 		this.marker.display();
 40 	gameSet.scene.popMatrix();
 41 }
 42 
 43 /**
 44 * Display function used to render this object.
 45 */
 46 TowerSelectionState.prototype.display = function(gameSet) {
 47 	gameSet.displayStatic();
 48 	
 49 	var prevValue;
 50 	if (!gameSet.scene.pickMode) {
 51 		prevValue = gameSet.scene.activeShader.getUniformValue("uAlphaScaling");
 52 		gameSet.scene.activeShader.setUniformsValues({uAlphaScaling: 0.5});
 53 		this.selectionPanelAppearance.apply();
 54 	}
 55 	var gameState = this;
 56 	for (var i = 0; i < this.positions.length; ++i) {
 57 		var position = this.positions[i];
 58 		var boardPosition = gameSet.board.getBoardCoordinates(position[0], position[1]);
 59 		
 60 		gameSet.scene.pushMatrix();
 61 			gameSet.scene.translate(boardPosition[0], boardPosition[1] + 0.01, boardPosition[2]);
 62 			gameSet.scene.registerNextPick({
 63 				row: position[0],
 64 				col: position[1],
 65 				onPick: function() {
 66 					gameState.onPickTower(gameSet, this.row, this.col);
 67 				}
 68 			});
 69 			this.selectionPanel.display();
 70 		gameSet.scene.popMatrix();
 71 	}
 72 	
 73 	if (!gameSet.scene.pickMode) {
 74 		gameSet.scene.activeShader.setUniformsValues({uAlphaScaling: prevValue});
 75 	}
 76 	gameSet.scene.clearPickRegistration();
 77 }

 79 /**
 80 * Function called to handle the tower selection.
 81 */
 82 TowerSelectionState.prototype.towerSet = function(gameSet, request) {
 83 	Connection.getTowers(gameSet, TowerSelectionState.prototype.updateTowers);
 84 	if (request == "Towers ready") {
 85 		gameSet.setState(new PlayGameState());
 86 		return;
 87 	}
 88 	
 89 	var requestData = JSON.parse(request);
 90 	var tower = requestData[0];
 91 	
 92 	switch(tower) {
 93 		case Connection.lightTower:
 94 			this.marker.setText("Select Light Tower");
 95 			this.pickingTower = Connection.light;
 96 			break;
 97 		case Connection.darkTower:
 98 			this.marker.setText("Select Dark Tower");
 99 			this.pickingTower = Connection.dark;
100 			break;
101 	}
102 	
103 	this.positions = requestData[1];
104 }
105 
106 /**
107 * Updates the games' towers information.
108 */
109 TowerSelectionState.prototype.updateTowers = function(gameSet, request) {
110 	var towerSchema = JSON.parse(request);
111 	var towers = [];
112 	for (var i = 0; i < towerSchema.length; ++i) {
113 		switch (towerSchema[i][0]) {
114 			case Connection.lightTower:
115 				towers.push(new LightTower(gameSet.scene));
116 				towers[towers.length - 1].setPosition(towerSchema[i][1], towerSchema[i][2]);
117 				break;
118 			case Connection.darkTower:
119 				towers.push(new DarkTower(gameSet.scene));
120 				towers[towers.length - 1].setPosition(towerSchema[i][1], towerSchema[i][2]);
121 				break;
122 		}
123 	}
124 	
125 	gameSet.setTowers(towers);
126 }

129 /**
130 * Function called when the user chooses a place to place a tower.
131 */
132 TowerSelectionState.prototype.onPickTower = function(gameSet, row, col) {
133 	var gameState = this;
134 	Connection.addTower(this.pickingTower, row, col, gameSet, function(gameSet, request) { gameState.towerAdded(gameSet, request);});
135 }

138 /**
139 * Function called when a tower is added.
140 */
141 TowerSelectionState.prototype.towerAdded = function(gameSet, request) {
142 	var gameState = this;
143 	Connection.setupTowers(gameSet, function (target, request) { gameState.towerSet(target, request); });
144 }
