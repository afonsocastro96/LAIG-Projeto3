1 /**
  2  * MySphere constructor.
  3  * @constructor
  4  * @param scene {CGFscene} The scene to which this sphere belongs.
  5  * @param radius {Float} Sphere radius.
  6  * @param slices {Integer} Number of divisions of each circle of the sphere.
  7  * @param stacks {Integer} Number of divisions of the sphere.
  8  */
  9  function MySphere(scene, radius, slices, stacks) {
 10  	CGFobject.call(this,scene);
 11 	
 12 	this.radius=radius;
 13 	this.slices=slices;
 14 	this.stacks=stacks;
 15 
 16  	this.initBuffers();
 17  };
 18 
 19  MySphere.prototype = Object.create(CGFobject.prototype);
 20  MySphere.prototype.constructor = MySphere;
 21 
 22 /**
 23  * Method in which the geometry of the cylinder is defined.
 24  */
 25  MySphere.prototype.initBuffers = function() {
 26 
 27 	this.indices = [];
 28  	this.vertices = [];
 29  	this.normals = [];
 30  	this.texCoords = [];
 31 
 32  	var dTheta = Math.PI / this.stacks;
 33  	var dPhi = 2 * Math.PI / this.slices;
 34 
 35 	for (var stack = 0; stack <= this.stacks; ++stack) {
 36 		for (var slice = 0; slice <= this.slices; ++slice) {
 37 			this.vertices.push(this.radius * Math.sin(stack * dTheta) * Math.cos(slice * dPhi), this.radius * Math.sin(stack * dTheta) * Math.sin(slice * dPhi), this.radius * Math.cos(stack * dTheta));
 38 			this.normals.push(Math.sin(stack * dTheta) * Math.cos(slice * dPhi), Math.sin(stack * dTheta) * Math.sin(slice * dPhi), Math.cos(stack * dTheta));
 39 			this.texCoords.push(slice/this.slices, stack/this.stacks);
 40 		}
 41 	}
 42 
 43 	for (var stack = 0; stack < this.stacks; ++stack) {
 44 		for (var slice = 0; slice < this.slices; ++slice) {
 45 			this.indices.push(stack * (this.slices + 1) + slice, (stack + 1) * (this.slices + 1) + slice, (stack + 1) * (this.slices + 1) + slice + 1);
 46 			this.indices.push(stack * (this.slices + 1) + slice, (stack + 1) * (this.slices + 1) + slice + 1, stack * (this.slices + 1) + slice + 1);
 47 		}
 48 	}
 49 
 50  	this.primitiveType = this.scene.gl.TRIANGLES;
 51  	this.initGLBuffers();
 52  };
