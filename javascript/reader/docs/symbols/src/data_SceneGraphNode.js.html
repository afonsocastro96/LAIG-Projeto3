1 /**
  2  * SceneGraph node constructor.
  3  * @constructor
  4  * @param id {string} Node identification.
  5  */
  6 function SceneGraphNode(id) {
  7     this.id = id;
  8     this.material = "null";
  9     this.texture = "clear";
 10     this.transformationMatrix = mat4.create();
 11     mat4.identity(this.transformationMatrix);
 12     this.descendants = [];
 13     this.animations = [];
 14 }
 15 
 16 SceneGraphNode.prototype = Object.create(Object.prototype);
 17 SceneGraphNode.prototype.constructor = SceneGraphNode;
 18 
 19 /**
 20  * Material setter.
 21  * @param material {string} Material to set.
 22  */
 23 SceneGraphNode.prototype.setMaterial = function(material) {
 24     this.material = material;
 25 }
 26 
 27 /**
 28  * Texture setter.
 29  * @param texture {string} Texture to set.
 30  */
 31 SceneGraphNode.prototype.setTexture = function(texture) {
 32      this.texture = texture;
 33 }
 34 
 35 /**
 36  * Adds a descendant to the descendants array.
 37  * @param descendant {string} Descendant to add.
 38  */
 39 SceneGraphNode.prototype.addDescendant = function(descendant) {
 40     this.descendants.push(descendant);
 41 }
 42 
 43 /**
 44  * Adds an animation to the animations array.
 45  * @param descendant {string} Animation to add.
 46  */
 47 SceneGraphNode.prototype.addAnimation = function(animation) {
 48     this.animations.push(animation);
 49 }
 50 /**
 51  * Applies a rotation to the node around the X axis.
 52  * @param rad {number} Rotation angle in radians.
 53  */
 54 SceneGraphNode.prototype.rotateX = function(rad) {
 55     mat4.rotateX(this.transformationMatrix, this.transformationMatrix, rad);
 56 }
 57 
 58 /**
 59  * Applies a rotation to the node around the Y axis.
 60  * @param rad {number} Rotation angle in radians.
 61  */
 62 SceneGraphNode.prototype.rotateY = function(rad) {
 63     mat4.rotateY(this.transformationMatrix, this.transformationMatrix, rad);
 64 }
 65 
 66 /**
 67  * Applies a rotation to the node around the Z axis.
 68  * @param rad {number} Rotation angle in radians.
 69  */
 70 SceneGraphNode.prototype.rotateZ = function(rad) {
 71     mat4.rotateZ(this.transformationMatrix, this.transformationMatrix, rad);
 72 }
 73 
 74 /**
 75  * Applies scaling to the node.
 76  * @param sx {number} X scale factor.
 77  * @param sy {number} Y scale factor.
 78  * @param sz {number} Z scale factor.
 79  */
 80 SceneGraphNode.prototype.scale = function(sx, sy, sz) {
 81     mat4.scale(this.transformationMatrix, this.transformationMatrix, vec3.fromValues(sx,sy,sz));
 82 }
 83 
 84 /**
 85  * Applies a translation to the node.
 86  * @param x {number} X translation value.
 87  * @param y {number} Y translation value.
 88  * @param z {number} Z translation value.
 89  */
 90 
 91 SceneGraphNode.prototype.translate = function(x, y, z) {
 92     mat4.translate(this.transformationMatrix, this.transformationMatrix, vec3.fromValues(x, y, z));
 93 }
