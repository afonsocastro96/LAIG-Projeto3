1 /**
  2  * BoardSelectionState constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which the BoardSelectionState belongs.
  5  */
  6 function BoardSelectionState(scene) {
  7 	GameState.call(this);
  8 }
  9 
 10 BoardSelectionState.prototype = Object.create(GameState.prototype);
 11 BoardSelectionState.prototype.constructor = BoardSelectionState;
 12 
 13 /**
 14 * BoardSelectionState initializer.
 15 */
 16 BoardSelectionState.prototype.init = function(gameSet) {
 17 	GameState.prototype.init.call(this, gameSet);
 18 	gameSet.setTowers([]);
 19 	
 20 	this.minorButton = new Marker(gameSet.scene);
 21 	this.minorButton.setText("Syrtis Minor");
 22 	this.majorButton = new Marker(gameSet.scene);
 23 	this.majorButton.setText("Syrtis Major");
 24 	
 25 	var gameState = this;
 26 	
 27 	this.minorPick = {
 28 		onPick: function() {
 29 			gameState.onPick(gameSet,Connection.minorBoard);
 30 		}
 31 	}
 32 	
 33 	this.majorPick = {
 34 		onPick: function() {
 35 			gameState.onPick(gameSet, Connection.majorBoard);
 36 		}
 37 	}
 38 }
 39 
 40 /**
 41 * Displays the state's HUD.
 42 */
 43 BoardSelectionState.prototype.displayHUD = function(gameSet) {
 44 	gameSet.scene.pushMatrix();
 45 		gameSet.scene.translate(0,1,-20);
 46 		gameSet.scene.scale(0.75, 0.75, 0.75);
 47 
 48 		gameSet.scene.registerNextPick(this.minorPick);
 49 		this.minorButton.display();
 50 	gameSet.scene.popMatrix();
 51 	
 52 	gameSet.scene.pushMatrix();
 53 		gameSet.scene.translate(0,-1,-20);
 54 		gameSet.scene.scale(0.75, 0.75, 0.75);
 55 		
 56 		gameSet.scene.registerNextPick(this.majorPick);
 57 		this.majorButton.display();
 58 	gameSet.scene.popMatrix();
 59 	
 60 	gameSet.scene.clearPickRegistration();
 61 }
 62 
 63 /**
 64 * Function called when a board is picked.
 65 */
 66 BoardSelectionState.prototype.onPick = function(gameSet, boardType) {
 67 	Connection.startgame(gameSet, this.gameStarted, boardType);
 68 }
 69 
 70 /**
 71 * Starts the mode selection state when the board is selected.
 72 */
 73 BoardSelectionState.prototype.gameStarted = function(gameSet, requestData) {
 74 	var boardSchema = JSON.parse(requestData);
 75 	var board = new GameBoard(gameSet.scene);
 76 	board.init(boardSchema);
 77 	
 78 	gameSet.setBoard(board);
 79 	gameSet.setState(new ModeSelectionState());
 80 }
