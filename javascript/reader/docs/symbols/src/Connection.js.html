1 /**
  2  * Handle prolog reply
  3  * @param data {Object} connection reply.
  4  * @param target {Object} object to be sent to the handler function.
  5  * @param handler {Function} handler function.
  6  */
  7 function handleReply(data, target, handler){
  8 	response=JSON.parse(data.target.response);
  9 	
 10 	if(handler != undefined){
 11 		handler(target, response.answer);
 12 	}
 13 }

 15 /**
 16  * Make request to prolog server
 17  * @param target {Object} object to be sent to the handler function.
 18  * @param request {String} request to make.
 19  * @param handler {Function} handler function.
 20  */
 21 function makeRequest(target, request, handler){
 22 	postGameRequest(request, function (data) { handleReply(data, target, handler); });	
 23 }
 24 
 25 /**
 26  * Make post request to prolog server.
 27  * @param requestString {String} request to make.
 28  * @param onSuccess {Function} function to be called on success.
 29  * @param onError {Function} function to be called on error.
 30  */
 31 function postGameRequest(requestString, onSuccess, onError)
 32 {
 33 	var request = new XMLHttpRequest();
 34 	request.open('POST', '../../game', true);
 35 
 36 	request.onload = onSuccess || function(data){console.log("Request successful. Reply: " + data.target.response);};
 37 	request.onerror = onError || function(){console.log("Error waiting for response");};
 38 
 39 	request.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded; charset=UTF-8');
 40 	request.send('requestString='+encodeURIComponent(requestString));
 41 }

261 /**
262  * Parse tile info.
263  * @param colourCode {Integer} tile colour code.
264  * @param shapeCode {Integer} tile shape code.
265  * @param scene {CGFscene} scene the tile belongs to.
266  * @return {BoardTile} returns board tile corresponding to codes.
267  */
268 Connection.parseTile = function(colourCode, shapeCode, scene) {
269 	if (colourCode == Connection.tileBlack) {
270 		if (shapeCode == Connection.tileCircle) {
271 			return new BlackCircleTile(scene);
272 		}
273 		if (shapeCode == Connection.tileSquare) {
274 			return new BlackSquareTile(scene);
275 		}
276 	}
277 	
278 	if (colourCode == Connection.tileWhite) {
279 		if (shapeCode == Connection.tileCircle) {
280 			return new WhiteCircleTile(scene);
281 		}
282 		if (shapeCode == Connection.tileSquare) {
283 			return new WhiteSquareTile(scene);
284 		}
285 	}
