1 /**
  2  * MyInterface constructor.
  3  * @constructor
  4  */
  5 function MyInterface() {
  6 	CGFinterface.call(this);
  7 };
  8 
  9 MyInterface.prototype = Object.create(CGFinterface.prototype);
 10 MyInterface.prototype.constructor = MyInterface;
 11 
 12 /**
 13  * Initializes the interface.
 14  * @param application {CGFapllication} Application this interface belongs to.
 15  */
 16 MyInterface.prototype.init = function(application) {
 17 	// call CGFinterface init
 18 	CGFinterface.prototype.init.call(this, application);
 19 	this.gui = new dat.GUI();
 20 };
 21 
 22 /**
 23  * Function to call when the scene is fully loaded.
 24  */
 25 MyInterface.prototype.onSceneLoaded = function(){
 26 	/* These options may be changed on the fly during the game */
 27 	var options = this.gui.addFolder("Change Options");
 28 	var scene = this.scene;
 29 	var myInterface = this;
 30 	options.add(this.scene, "currentCamera", Object.keys(this.scene.cameraPositions))
 31 		.name("Current Camera")
 32 		.onFinishChange(function(value) {
 33 			scene.setCamera(value);
 34 		}
 35 	);
 36 	
 37 	options.add(this.scene, "currentTheme", Object.keys(this.scene.themes))
 38 		.name("Theme")
 39 		.onFinishChange(function(value) {
 40 			scene.setTheme(value);
 41 		}
 42 	);
 43 	
 44 	options.add(this.scene.gameSet, "turnDuration", this.scene.gameSet.minTurnDuration, this.scene.gameSet.maxTurnDuration)
 45 		.name("Turn Duration");
 46 	this.interfaceLoaded = true;
 47 }
