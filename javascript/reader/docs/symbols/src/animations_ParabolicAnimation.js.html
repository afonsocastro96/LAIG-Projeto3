1 /**
  2  * ParabolicAnimation constructor.
  3  * @constructor
  4  * @param start {Array(3)} Initial position.
  5  * @param end {Array(3)} Final position.
  6  * @param elevation {Float} Elevation before arc.
  7  * @param peak {Float} Arc height.
  8  * @param span {Integer} Animation duration in miliseconds.
  9  */
 10 function ParabolicAnimation(start, end, elevation, peak, span) {
 11 	this.start = start;
 12 	this.end = end;
 13 	this.elevation = elevation;
 14 	this.peak = peak;
 15 	this.span = span;
 16 	this.init();
 17 }
 18 
 19 ParabolicAnimation.prototype = Object.create(Object.prototype);
 20 ParabolicAnimation.prototype.constructor = ParabolicAnimation;
 21 
 22 /**
 23  * Init animation values
 24  */
 25 ParabolicAnimation.prototype.init = function() {
 26 	var vector = [this.end[0] - this.start[0], this.end[1] - this.start[1], this.end[2] - this.start[2]];
 27 	
 28 	this.radius = this.vectorMagnitude(vector) / 2;
 29 	
 30 	this.angle = Math.acos(vector[0]/(this.radius * 2));
 31 	if (vector[2] > 0) {
 32 		this.angle = -this.angle;
 33 	}
 34 	
 35 	var parabolaDistance = (Math.pow(this.radius,3)*Math.asinh(2*this.peak/this.radius) +
 36 		2*this.peak*Math.sqrt(4*Math.pow(this.radius,2)*Math.pow(this.peak,2) + Math.pow(this.radius,4)))/
 37 		(2*this.radius*this.peak);
 38 	
 39 	var totalDistance = parabolaDistance + 2 * this.elevation;
 40 	
 41 	var velocity = totalDistance / this.span;
 42 	
 43 	this.controlPointsTime = [0, this.elevation / velocity, (this.elevation + parabolaDistance) / velocity, this.span];
 44 	this.controlPointsSpan = [];
 45 	for (var i = 0; i < this.controlPointsTime.length - 1; ++i) {
 46 		this.controlPointsSpan.push(this.controlPointsTime[i+1] - this.controlPointsTime[i]);
 47 	}
 48 	
 49 	this.controlPoints = [];
 50 	this.controlPoints.push(this.start);
 51 	this.controlPoints.push([this.start[0], this.start[1] + this.elevation, this.start[2]]);
 52 	this.controlPoints.push([this.end[0], this.end[1] + this.elevation, this.end[2]]);
 53 	this.controlPoints.push(this.end);
 54 	
 55 	this.center = [];
 56 	for (var i = 0; i < 3; ++i) {
 57 		this.center.push((this.controlPoints[1][i] + this.controlPoints[2][i]) / 2);
 58 	}
 59 	
 60 	var animation = this;
 61 	
 62 	this.pointFunction = [];
 63 	this.pointFunction.push(function (t) {
 64 		return animation.linearFunction(t, animation.controlPointsSpan[0], animation.controlPoints[0], animation.controlPoints[1]);
 65 	});
 66 	this.pointFunction.push(function (t) {
 67 		return animation.parabolicFunction(t, animation.controlPointsSpan[1], animation.center,animation.radius,animation.peak, animation.angle);
 68 	});
 69 	this.pointFunction.push(function(t) {
 70 		return animation.linearFunction(t, animation.controlPointsSpan[2], animation.controlPoints[2], animation.controlPoints[3]);
 71 	});
 72 }
 73 
 74 /**
 75  * Calculate the position in the animation.
 76  * @param t {Integer} Time (in miliseconds) occurred since the beginning of the animation.
 77  * @return {Array(3)} returns the position in the animation.
 78  */
 79 ParabolicAnimation.prototype.getPoint = function(t) {
 80 	if (t <= 0) {
 81         return this.start;
 82 	}
 83 	
 84 	if (t >= this.span) {
 85 		return this.end;
 86 	}
 87 	
 88 	var index;
 89     for (index = this.controlPointsTime.length - 1; index > 0; --index)
 90         if (this.controlPointsTime[index] < t)
 91             break;
 92 
 93 	return this.pointFunction[index](t - this.controlPointsTime[index]);
 94 }

 96 /**
 97  * Calculate vector magnitude
 98  * @param vector {Array(3)} 3D vector.
 99  * @return {Float} returns vector magnitude.
100  */
101 ParabolicAnimation.prototype.vectorMagnitude = function(vector) {
102 	return Math.sqrt(
103 		Math.pow(vector[0], 2) +
104 		Math.pow(vector[1], 2) +
105 		Math.pow(vector[2], 2));
106 }

108 /**
109  * Calculate position in a linear movement.
110  * @param t {Integer} Time occurred (in miliseconds) since the beginning of the movement.
111  * @param span {Integer} Movement duration (in miliseconds).
112  * @param start {Array(3)} Initial position.
113  * @param end {Array(3)} Final position.
114  * @return {Array(3)} returns position in the movement.
115  */
116 ParabolicAnimation.prototype.linearFunction = function(t, span, start, end) {
117 	var point = [];
118 	
119 	for (var i = 0; i < start.length; ++i) {
120 		point.push(start[i] + (end[i] - start[i]) * t / span);
121 	}
122 	
123 	return point;
124 }
125 
126 /**
127  * Calculate position in a parabolic movement.
128  * @param t {Integer} Time occurred (in miliseconds) since the beginning of the movement.
129  * @param span {Integer} Movement duration (in miliseconds).
130  * @param center {Array(3)} Center of the parabola base.
131  * @param radius {Float} Radius of the parabola base.
132  * @param peak {Float} Parabola height.
133  * @param angle {Float} Parabola angle with the x axis.
134  * @return {Array(3)} returns position in the movement.
135  */
136 ParabolicAnimation.prototype.parabolicFunction = function(t, span, center, radius, peak, angle) {
137 	var x = -radius + 2 * radius * t / span;
138 	
139 	var y = -peak * (x - radius) * (x + radius) / Math.pow(radius, 2);
140 	
141 	var transformation = mat4.create();
142 	
143 	mat4.translate(transformation, transformation, vec3.fromValues(center[0], center[1], center[2]));
144 	mat4.rotateY(transformation, transformation, angle);
145 	mat4.translate(transformation, transformation, vec3.fromValues(x, y, 0));
146 	
147 	var point = vec3.fromValues(0, 0, 0);
148 	vec3.transformMat4(point, point, transformation);
149 	
150 	return point;
151 }
