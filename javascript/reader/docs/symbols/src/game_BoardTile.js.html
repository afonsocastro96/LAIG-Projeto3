1 /**
  2  * BoardTile constructor.
  3  * @constructor
  4  * @param scene {CGFScene} The scene to which the DarkTower belongs.
  5  */
  6 function BoardTile(scene) {
  7     this.scene = scene;
  8 	this.alphaScaling = 1.0;
  9 	this.setTransparency(false);
 10 }
 11 
 12 BoardTile.prototype = Object.create(CGFobject.prototype);
 13 BoardTile.prototype.constructor = BoardTile;
 14 
 15 /**
 16 * Display function used to render this object.
 17 */
 18 BoardTile.prototype.display = function() {
 19 	this.displayFunction();
 20 }
 21 
 22 BoardTile.prototype.defaultDisplay = function() {
 23 	//do nothing
 24 }
 25 
 26 /**
 27 * Display function used when a move is possible with this tile.
 28 */
 29 BoardTile.prototype.transparentDisplay = function() {
 30 	var prevValue;
 31 	if (!this.scene.pickMode) {
 32 		prevValue = this.scene.activeShader.getUniformValue("uAlphaScaling");
 33 		this.scene.activeShader.setUniformsValues({uAlphaScaling: this.alphaScaling});
 34 	}
 35 	this.defaultDisplay();
 36 	
 37 	if (!this.scene.pickMode) {
 38 		this.scene.activeShader.setUniformsValues({uAlphaScaling: prevValue});
 39 	}
 40 }
 41 
 42 /**
 43 * Sets the transparency of the tile, according to the fact that there is a move possible with this tile or not
 44 */
 45 BoardTile.prototype.setTransparency = function(bTransparent) {
 46 	if (bTransparent) {
 47 		this.displayFunction = this.transparentDisplay;
 48 	}
 49 	else {
 50 		this.displayFunction = this.defaultDisplay;
 51 	}
 52 }
 53 
 54 /**
 55 * Sets the alpha scaling for when a move is possible for this tile
 56 */
 57 BoardTile.prototype.setAlphaScaling = function(alphaScaling) {
 58 	this.alphaScaling = alphaScaling;
 59 }
